\newcommand{\blockCol}[2]{
    \begin{scope}[shift={(#1)}]
        \setcounter{len}{0}
        \foreach \num/\col in {#2}{
            \stepcounter{len}
        }
        \foreach \num/\col [count=\x] in {#2}{ 
            \draw[densely dotted] (\x,0) -- (\x,1);
            \node[text=\col] at (\x-0.5,0.5) {\num}; 
        }
        \draw[line width=0.25mm] (0,0) rectangle (\value{len},1);

    \end{scope}
}

\newcommand{\FADESEGMENTS}[1]{
  \foreach \x/\ystart/\yend in {#1} {
    \coordinate (A) at (\x,\ystart);
    \tikzmath{\z = \ystart + ((\yend - \ystart) / 4 * 3);}
    \coordinate (B) at (\x,\z);
    \foreach \i in {0, 1, ..., 9} {
      \tikzmath{int \j; \j = \i + 1; \k = (\i + .5) / 10 * 100;}
      \draw[color=black!\k!lightgray] ($(B)!\i/10!(A)$) -- ($(B)!\j/10!(A)$);
    }
    \draw[lightgray] (B) -- (\x,\yend-0.075);
    \draw[black, fill=black] (A) circle (0.075);
    \draw[lightgray, fill=none] (\x,\yend) circle (0.075);
  }}
